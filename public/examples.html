01. TRUE RANDOMNESS 
// Generate a True Random Number (0-255) using Hardware Entropy
function getTrueRandom() {
    const array = new Uint8Array(1);
    window.crypto.getRandomValues(array);
    return array[0];
}
02. GENERATE TOKEN 
// Create a Secure Session Key (Hex format)
function generateKey() {
    return 'sk_live_' + Array.from(
        window.crypto.getRandomValues(new Uint8Array(16)),
        byte => byte.toString(16).padStart(2, '0')
    ).join('');
}
