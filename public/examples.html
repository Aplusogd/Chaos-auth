01. TRUE RANDOMNESS 
// Generate a True Random Number (0-255)
function getTrueRandom() {
    const array = new Uint8Array(1);
    window.crypto.getRandomValues(array);
    return array[0];
}
02. GENERATE TOKEN 
// Create a Secure Session Key
function generateKey() {
    return 'sk_live_' + Array.from(
        window.crypto.getRandomValues(new Uint8Array(16)),
        byte => byte.toString(16).padStart(2, '0')
    ).join('');
}
03. CHAOS DECISION 
// 50/50 Choice based on Entropy
function chaosFlip() {
    const r = new Uint8Array(1);
    window.crypto.getRandomValues(r);
    return r[0] % 2 === 0 ? "ACTION_A" : "ACTION_B";
}
04. GLITCH TEXT EFFECT 
// Corrupt text visually
function glitchText(text) {
    const chars = "!@#$%^&*<>?/";
    return text.split('').map(c => 
        Math.random() > 0.9 ? chars[Math.floor(Math.random()*chars.length)] : c
    ).join('');
}
05. SHA-256 HASH 
// Hash data (Requires CryptoJS)
function secureHash(input) {
    return CryptoJS.SHA256(input).toString();
}
